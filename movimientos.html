def movimientos():
    #importar libreria opencv
    import cv2
    #guardamos la imagen en una var
    imagen=cv2.imread("gato_3.jpg")
    imoriginal=imagen.copy()#Creo una copia de la imagen

    while True:#ciclo infinito
        cv2.imshow("imagen inicial",imagen)
        tecla=cv2.waitKey(0)#Espera infita hasta la presion de una tecla

        if tecla==ord('d'): #D gira a la derecha
            imagen=cv2.rotate(imagen,cv2.ROTATE_90_CLOCKWISE)#rota a 90 grados en sentido del reloj
            print("imagen rotando a la derecha",tecla)
        elif tecla==ord('a'): #A gira a la izquierda
            imagen=cv2.rotate(imagen,cv2.ROTATE_90_COUNTERCLOCKWISE)#90 grados al contrario del reloj
            print("imagen rotando  al izquierda",tecla)
        elif tecla==ord('w'): #gira 180 grados
            imagen=cv2.rotate(imagen,cv2.ROTATE_180)#Giro de 180 grados
            print("Se presiono la tecla ",tecla)
        elif tecla==ord('s'): #S Tecla para salir
            print("Se presiono la tecla ",tecla)
            break
if __name__=="__main__":
    movimientos()

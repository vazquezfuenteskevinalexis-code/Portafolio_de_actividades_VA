import cv2
import pytesseract

pytesseract.pytesseract.tesseract_cmd=r"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tesseract.exe"

import os
os.environ['TESSDATA_PREFIX'] = r"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tessdata"


#lista de palcas aceptadas en el estacionamiento del CBTIS144
placas_autorizdas=['ASD434','FRG-256','PLF675','PCL432']

#Cargar placa a leer
placa=cv2.imread("placas.jpg")
#placa_nueva=cv2.resize(placa,(800,400))

#Convertir a escala de grises
placa_gris=cv2.cvtColor(placa,cv2.COLOR_BGR2GRAY)

#Separamos el fondo con el texto
placa_res=cv2.threshold(placa_gris,120,255,cv2.THRESH_BINARY)[1]

#Configurar OCR
texto_placa=pytesseract.image_to_string(placa_res,lang="spa")

#Limpiar la cadena de texto.
texto_limpio=texto_placa.strip().replace(" ","_")

#Buscar la placa en la lista autorizada
if texto_limpio in placas_autorizdas:
    print("ACESO PERMITIDO!!,\n Bienvenido al estacionamiento del CBTIS144")
else:
    print("ACCESO DENEGADO!!,\n La placa no esta autorizada en la lista.")

print(texto_limpio)
#Imprimimos el titulo
print("Texto detectado")
print(texto_placa)

cv2.imshow("Imagen analizada",placa_res)
cv2.waitKey(0)
cv2.destroyAllWindows()

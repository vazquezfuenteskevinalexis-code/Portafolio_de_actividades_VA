def zoom():
    from importlib import reload

    import cv2
    #Cargo la imagen en una variable.
    img_original = cv2.imread('gato_3.jpg')
    #Creando la variable para el valor del zoom original.
    zoom=1.0

    while True:
        img=cv2.resize(img_original,None,fx=zoom,fy=zoom)
        cv2.imshow("Imagen nueva",img)
        tecla=cv2.waitKey(0)
        if tecla==ord('+'):
            zoom=zoom*1.2
        elif tecla==ord('-'):
            zoom=zoom*0.1
        elif tecla==ord('o'):
             zoom=1.0
        elif tecla==ord('x'):
             break

    cv2.destroyAllWindows()
if __name__ == '__main__':
    zoom()

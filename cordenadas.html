def recortes():
    import cv2 #importa libreria opecv

    #cargar imagen
    gato=cv2.imread('../gato..jpg')

    while True:

        recorte=gato[40:390, 69:390]#alto y ancho
        cv2.imshow('imagen original', gato)
        cv2.imshow('recorte', recorte)
        tecla=cv2.waitKey(0)#guarda en al variable tecla presiona

        if tecla==ord('o'):
                cv2.imshow('imagen original', gato)

        elif tecla == ord('y'):
            cv2.imshow('recorte', recorte)

        elif tecla == ord('x'):
            cv2.destroyAllWindows()
            break
"""_main_ es el nombre del archivo  cuando se ejecuta en si mismo"""
if __name__ == '__main__':
    recortes()
